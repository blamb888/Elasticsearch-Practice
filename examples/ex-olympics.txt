DELETE _ingest/pipeline/olympics

PUT _ingest/pipeline/olympics
{
  "processors": [
    {
      "pipeline": {
        "name": "lang-per-field"
      }
    },
    {
      "script": {
        "lang": "painless",
        "source": "if (['ja', 'ko', 'zh'].contains(ctx.contents.language)) { ctx.contents.cjk = ctx.contents.default }"
      }
    }
  ]
}
